% watermark.tex
\usepackage{everypage} % For running code on every page
\usepackage{tikz}      % For drawing overlays
\usepackage{xcolor}    % For color options
\usepackage{graphicx}  % For including graphics

\newcommand{\EvenPageWatermark}{%
  \ifodd\value{page}% 
    % Do nothing on odd pages
  \else%
    \begin{tikzpicture}[remember picture,overlay]%
      \node[rotate=0, opacity=0.1] at (current page.center) {%
        \includegraphics[width=0.5\paperwidth]{watermark.png}%
      };%
    \end{tikzpicture}%
  \fi%
}%

\AddEverypageHook{\EvenPageWatermark}%
